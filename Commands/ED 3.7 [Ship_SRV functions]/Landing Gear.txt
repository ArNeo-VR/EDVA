    Begin Condition : [{CMDSEGMENT:2}] Equals 'down' OR [{CMDSEGMENT:2}] Equals 'up' OR [{CMDSEGMENT:0}] Equals 'deploy' OR [{CMDSEGMENT:0}] Equals 'retract'

End Condition
