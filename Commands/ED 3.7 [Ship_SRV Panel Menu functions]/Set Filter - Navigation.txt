= Add trigger word to the 'When I say'-command (above)

= Delete the one's you don't want enabled.

Set Boolean [auto-panel-select] to True
# Set [ALL] variables Settings to false
Set Boolean [Stars] to False
Set Boolean [Asteroid clusters] to False

Set Boolean [Landfall planets and Moons] to False

# Set Selections
Begin Condition : [{CMDSEGMENT:1}] Equals 'All' OR ([{CMDSEGMENT:1}] Equals 'All' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Stars] to True
    Set Boolean [Asteroid clusters] to True
    Set Boolean [Planet and Moons] to True
    Set Boolean [Landfall planets and Moons] to True
    Set Boolean [Settlements] to True
    Set Boolean [Stations] to True
    Set Boolean [Fleet Carriers] to True
    Set Boolean [Points of Interest] to True
    Set Boolean [Signal sources] to True
    Set Boolean [Systems] to True
Else If : [{CMDSEGMENT:1}] Equals 'Default' OR ([{CMDSEGMENT:1}] Equals 'Default' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Stars] to True
    Set Boolean [Planet and Moons] to True
    Set Boolean [Asteroid clusters] to True
    Set Boolean [Landfall planets and Moons] to True
    Set Boolean [Settlements] to True
    Set Boolean [Stations] to True
    Set Boolean [Points of Interest] to True
    Set Boolean [Signal sources] to True
Else If : [{CMDSEGMENT:1}] Equals 'Missions' OR ([{CMDSEGMENT:1}] Equals 'Missions' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Landfall planets and Moons] to True
    Set Boolean [Settlements] to True
    Set Boolean [Stations] to True
    Set Boolean [Systems] to True
    --  Malic, navigation filter names ##################################################################
Else If : ([{CMDSEGMENT:1}] Equals 'Standard' AND [{CMDSEGMENT:2}] Has Not Been Set) OR [{CMDSEGMENT:1}] Equals 'Default' OR ([{CMDSEGMENT:1}] Equals 'Standard' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'Default' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Stars] to True
    Set Boolean [Planet and Moons] to True
    Set Boolean [Landfall planets and Moons] to True
    Set Boolean [Settlements] to True
    Set Boolean [Stations] to True
    Set Boolean [Points of Interest] to True
    Set Boolean [Signal sources] to True
    Set Boolean [Systems] to True
Else If : [{CMDSEGMENT:1}] Equals 'Settlements' OR ([{CMDSEGMENT:1}] Equals 'Settlements' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Settlements] to True
    Set Boolean [Stations] to True
Else If : [{CMDSEGMENT:1}] Equals 'stations' OR ([{CMDSEGMENT:1}] Equals 'stations' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Stations] to True
Else If : [{CMDSEGMENT:1}] Equals 'Fleet Carriers' OR ([{CMDSEGMENT:1}] Equals 'Fleet Carriers' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Fleet Carriers] to True
Else If : [{CMDSEGMENT:1}] Equals 'unknown' OR [{CMDSEGMENT:1}] Equals 'anomaly' OR ([{CMDSEGMENT:1}] Equals 'unknown' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'anomaly' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Points of Interest] to True
    Set Boolean [Signal sources] to True
Else If : [{CMDSEGMENT:1}] Equals 'signals' OR [{CMDSEGMENT:1}] Equals 'scanner' OR [{CMDSEGMENT:1}] Equals 'signal source' OR ([{CMDSEGMENT:1}] Equals 'signals' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'scanner' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'signal source' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Signal sources] to True
Else If : [{CMDSEGMENT:1}] Equals 'systems' OR [{CMDSEGMENT:1}] Equals 'next system only' OR [{CMDSEGMENT:1}] Equals 'system' OR ([{CMDSEGMENT:1}] Equals 'systems' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'next system only' AND [{CMDSEGMENT:2}] Equals 'Filters') OR ([{CMDSEGMENT:1}] Equals 'system' AND [{CMDSEGMENT:2}] Equals 'Filters')
    Set Boolean [Systems] to True
End Condition - Exit when condition not met

>> Execute, Set Filters

>> Back to main viewer
Begin Boolean Compare : [auto-panel-select] Equals True
    
    Start Loop While : ([{TXT:Status gui focus}] Does Not Equal 'none' AND [{TXT:Status gui focus}] Does Not Equal 'external panel')
        Pause 0,06 seconds
        Press variable key(s) [edUI_Back] and hold for 0,05 seconds and release
    End Loop
    
    >> Goto [navigation panel] Ship/SRV
    Execute command, '[External; Navigation; Nav; Target] Panel' (and wait until it completes)
    
    DISABLED - Begin Text Compare : [{TXT:Status gui focus}] Does Not Equal 'external panel'
        DISABLED - Start Loop : Repeat 3 Times
            DISABLED - Pause 0,06 seconds
            DISABLED - Press variable key(s) [edUI_Back] and hold for 0,05 seconds and release
        DISABLED - End Loop
        
        >> Goto [navigation panel] Ship/SRV
        DISABLED - Begin Text Compare : [{TXT:Status vehicle}] Equals 'Ship'
            DISABLED - Press variable key(s) [edFocusLeftPanel] and hold for 0,05 seconds and release
        DISABLED - Else If Text Compare : [{TXT:Status vehicle}] Equals 'SRV'
            DISABLED - Press variable key(s) [edFocusLeftPanel_Buggy] and hold for 0,05 seconds and release
        DISABLED - End Condition
        
    DISABLED - End Condition
    Pause 0,09 seconds
    
    ##############################################################
    >> When in different [tab] move back to the left (variable [nav-tab])
    Start Loop While : [nav-tab] Is Greater Than 0
        Press variable key(s) [edLeftThrustButton] and hold for 0,05 seconds and release
        Pause 0,06 seconds
        Set small int (condition) [nav-tab] value as decremented by 1
    End Loop
    ##############################################################
End Condition

    Press variable key(s) [edUI_Up] and hold for 0,05 seconds and release
    Pause 0,06 seconds

>> reset filters
Begin Condition : ([{CMDSEGMENT:0}] Starts With 'Set' AND [{CMDSEGMENT:1}] Does Not Equal '') OR ([{CMDSEGMENT:0}] Equals 'Set' AND [{CMDSEGMENT:1}] Does Not Equal '')
    Begin Boolean Compare : [filter] Equals True
        Press variable key(s) [edUI_Down] and hold for 0,05 seconds and release
        Set Boolean [filter] to False
        Pause 0,04 seconds
        Pause 0,2 seconds
        Press variable key(s) [edUI_Select] and hold for 0,05 seconds and release
        Pause 0,2 seconds
        Press variable key(s) [edUI_Select] and hold for 0,05 seconds and release
    Else
        Pause 0,2 seconds
        Press variable key(s) [edUI_Select] and hold for 0,05 seconds and release
        Pause 0,5 seconds
    End Condition
    Pause 0,2 seconds
    DISABLED - Press variable key(s) [edUI_Up] and hold for 0,05 seconds and release
    DISABLED - Pause 0,5 seconds
End Condition
DISABLED - Press variable key(s) [edUI_Select] and hold for 0,05 seconds and release
DISABLED - Pause 0,5 seconds
DISABLED - Press Space key and hold for 0,05 seconds and release
